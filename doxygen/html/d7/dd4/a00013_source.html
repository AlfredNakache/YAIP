157 void DisplayINI(YAIP::YAIP &IniParser, std::string &Comment)
158 {
159  std::cout << "----- " << Comment << " -----" << std::endl;
160 
161  YAIP::tVectorString SectionList = IniParser.SectionListGet();
162  for (YAIP::tVectorString::iterator LoopSection = SectionList.begin(); LoopSection != SectionList.end(); ++LoopSection)
163  {
164  std::string Section = *LoopSection;
165  if (0 < Section.length())
166  {
167  std::cout << "[" << Section << "]" << std::endl;
168  }
169 
170  YAIP::tVectorString KeyList = IniParser.SectionKeyListGet(Section);
171  int SectionCount = 0;
172  for (YAIP::tVectorString::iterator LoopKey = KeyList.begin(); LoopKey != KeyList.end(); ++LoopKey)
173  {
174  std::string Default = "Default";
175  std::string Key = *LoopKey;
176  std::string Value = IniParser.SectionKeyValueGet(Section, Key, Default);
177  std::cout << Key << "=" << Value << std::endl;
178  SectionCount++;
179  }
180  }
181 }
